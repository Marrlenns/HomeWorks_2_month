								SQL запросы


CREATE - команда для создания новаой таблицы данных

INSERT - команда для добавления данных в таблицу

SELECT - команда для получения данных из выбранной таблицы. Следующей командой можно вывести все данные из таблицы:
															SELECT * FROM <table_name>
UPDATE - команда для обновления данных таблицы.

DROP - команда для удаления строк из таблицы, всей таблицы целиком или целой базы данных.
